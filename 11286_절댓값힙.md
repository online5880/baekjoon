### Priority Queue

```python
from collections import defaultdict, deque
from queue import PriorityQueue
import sys
input = sys.stdin.readline

N = int(input())

pq = PriorityQueue()

for i in range(N):
    request = int(input())
    if request != 0:
        pq.put((abs(request),request))
    else:
        if pq.empty():
            print(0)
        else:
            print(pq.get()[1])
```

### heapq

```python
import heapq
import sys
input = sys.stdin.readline

N = int(input())

hq = []

for i in range(N):
    request = int(input())
    if request != 0:
        heapq.heappush(hq,(abs(request),request))
    else:
        if not hq:
            print(0)
        else:
            print(heapq.heappop(hq)[1])
```
